function in_check {

 RF="lanmp_wdcp-2-5.$arch.rpm";

 RF="lanmp_wdcp-2-5.el6.$arch.rpm";
 if [ $arch == "x86_64" ];then

 fi
 yum install -y mhash-devel libmcrypt-devel
else
 echo

 exit
fi
wget -c http://dl.wdlinux.cn:5180/rpms/$RF

I_F=$?
